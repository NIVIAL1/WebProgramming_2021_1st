            function capture(e){
                var obj = e.currentTarget;
                var tagName = obj.tagName;
                div.innerHTML += "<br> Capture 단계 : " + tagName + "태그 : " + obj + "이벤트 : " + e.type;
            }
            function bubble(e){
                var obj = e.currentTarget;
                var tagName = obj.tagName;
                div.innerHTML += "<br> Bubble 단계 : " + tagName + "태그 : " + obj + "이벤트 : " + e.type;
